# 旅游网站酒店价格爬虫使用说明 (CR008)

## 📋 概述

旅游网站酒店价格爬虫（TravelCrawler）是一个专业的酒店价格监控工具，支持爬取携程、去哪儿、飞猪三大主流旅游平台的酒店信息。该爬虫提供实时价格监控、智能分析和数据导出功能。

## 🚀 快速开始

### 安装依赖

```bash
pip install -r requirements.txt
```

### 基本使用

```python
from crawlers.travel.travel_crawler import TravelCrawler

# 创建爬虫实例
crawler = TravelCrawler()

# 搜索酒店
result = crawler.run(
    city="北京",
    check_in="2024-12-01",
    check_out="2024-12-07",
    platform="all",  # 搜索所有平台
    max_hotels=50    # 限制最大酒店数
)

# 导出数据
filepath = crawler.export_data(result)
print(f"数据已导出到: {filepath}")
```

### 通过主程序使用

```bash
# 查看所有爬虫
python main.py --list

# 运行旅游爬虫
python main.py --run travel --city 上海 --check_in 2024-12-01 --check_out 2024-12-07

# 运行演示模式
python main.py --demo
```

## 📊 数据字段

爬虫返回的数据包含以下字段：

### 酒店基本信息
- `hotel_id` - 酒店ID
- `hotel_name` - 酒店名称
- `platform` - 平台名称
- `city` - 城市
- `address` - 详细地址
- `star_rating` - 星级 (0-5)
- `user_rating` - 用户评分 (0-5)
- `review_count` - 评论数量

### 价格信息
- `lowest_price` - 最低价格 (元)
- `original_price` - 原价 (元)
- `discount` - 折扣率 (%)
- `room_type` - 房型
- `room_name` - 房间名称

### 设施服务
- `breakfast` - 早餐信息
- `wifi` - WiFi信息
- `parking` - 停车信息
- `facilities` - 设施列表 (逗号分隔)
- `tags` - 标签 (逗号分隔)

### 其他信息
- `check_in` - 入住时间
- `check_out` - 离店时间
- `phone` - 联系电话
- `latitude` - 纬度
- `longitude` - 经度
- `source_url` - 来源URL
- `crawl_time` - 爬取时间

## 📋 使用示例

### 示例1：酒店价格监控

```python
import schedule
import time
from datetime import datetime
from crawlers.travel.travel_crawler import TravelCrawler

def monitor_hotel_prices():
    """定时监控酒店价格"""
    crawler = TravelCrawler()
    
    # 监控多个城市
    cities = ["北京", "上海", "杭州"]
    
    for city in cities:
        print(f"监控{city}酒店价格...")
        result = crawler.run(city=city, max_hotels=30)
        
        # 分析价格变化
        analysis = result["analysis"]
        price_stats = analysis["price_stats"]
        
        print(f"{city}酒店价格统计:")
        print(f"  最低价: ¥{price_stats['min']}")
        print(f"  最高价: ¥{price_stats['max']}")
        print(f"  平均价: ¥{price_stats['average']}")
        
        # 导出数据
        timestamp = datetime.now().strftime("%Y%m%d_%H%M")
        filename = f"{city}_酒店价格_{timestamp}.xlsx"
        crawler.export_data(result, filename)

# 每天上午10点运行
schedule.every().day.at("10:00").do(monitor_hotel_prices)

while True:
    schedule.run_pending()
    time.sleep(60)
```

### 示例2：旅游规划助手

```python
from crawlers.travel.travel_crawler import TravelCrawler

def travel_planner(destination, budget, dates):
    """旅游规划助手"""
    crawler = TravelCrawler()
    
    # 搜索酒店
    result = crawler.run(
        city=destination,
        check_in=dates["check_in"],
        check_out=dates["check_out"],
        max_hotels=100
    )
    
    # 按预算过滤
    affordable_hotels = [
        hotel for hotel in result["data"]
        if hotel["lowest_price"] <= budget
    ]
    
    # 按评分排序
    affordable_hotels.sort(key=lambda x: x["user_rating"], reverse=True)
    
    print(f"在{budget}元预算内找到{len(affordable_hotels)}家酒店:")
    for i, hotel in enumerate(affordable_hotels[:5], 1):
        print(f"{i}. {hotel['hotel_name']}")
        print(f"   价格: ¥{hotel['lowest_price']}")
        print(f"   评分: {hotel['user_rating']}/5.0")
        print(f"   地址: {hotel['address']}")
        print()
    
    return affordable_hotels

# 使用示例
destination = "杭州"
budget = 500
dates = {"check_in": "2024-12-01", "check_out": "2024-12-03"}
travel_planner(destination, budget, dates)
```

### 示例3：竞品价格分析

```python
import pandas as pd
from crawlers.travel.travel_crawler import TravelCrawler

def competitor_analysis(hotel_name, cities):
    """竞品酒店价格分析"""
    crawler = TravelCrawler()
    all_data = []
    
    for city in cities:
        result = crawler.run(city=city, max_hotels=50)
        
        # 查找目标酒店
        target_hotels = [
            hotel for hotel in result["data"]
            if hotel_name.lower() in hotel["hotel_name"].lower()
        ]
        
        for hotel in target_hotels:
            hotel["search_city"] = city
            all_data.append(hotel)
    
    # 转换为DataFrame分析
    if all_data:
        df = pd.DataFrame(all_data)
        
        # 按城市分组分析
        city_stats = df.groupby("search_city").agg({
            "lowest_price": ["min", "max", "mean", "count"],
            "user_rating": "mean"
        }).round(2)
        
        print("竞品酒店价格分析:")
        print(city_stats)
        
        # 导出分析报告
        output_file = f"{hotel_name}_竞品分析.xlsx"
        df.to_excel(output_file, index=False)
        print(f"分析报告已导出到: {output_file}")
    
    return all_data

# 使用示例
competitor_analysis("希尔顿", ["北京", "上海", "广州", "杭州"])
```

### 缓存策略

```python
import pickle
from datetime import datetime, timedelta

def cached_search(city, cache_hours=24):
    """带缓存的搜索"""
    cache_file = f"cache/{city}_hotels.pkl"
    
    # 检查缓存
    try:
        with open(cache_file, "rb") as f:
            cached_data, cache_time = pickle.load(f)
            
            # 检查缓存是否过期
            if datetime.now() - cache_time < timedelta(hours=cache_hours):
                print(f"使用缓存数据: {city}")
                return cached_data
    except:
        pass
    
    # 执行搜索
    crawler = TravelCrawler()
    result = crawler.run(city=city)
    
    # 保存缓存
    with open(cache_file, "wb") as f:
        pickle.dump((result, datetime.now()), f)
    
    return result
```

```python
# 与机器学习集成
from sklearn.cluster import KMeans
import numpy as np

def cluster_hotels(hotels_data):
    """酒店聚类分析"""
    # 提取特征
    features = []
    for hotel in hotels_data:
        features.append([
            hotel["lowest_price"],
            hotel["user_rating"] * 100,
            hotel["star_rating"] * 20
        ])
    
    # K-means聚类
    kmeans = KMeans(n_clusters=3)
    clusters = kmeans.fit_predict(features)
    
    # 分析聚类结果
    for i, hotel in enumerate(hotels_data):
        hotel["cluster"] = int(clusters[i])
    
    return hotels_data
```
